<p><b>User ID:</b> <span id="uid">—</span></p>
<p><b>Username:</b> <span id="uname">—</span></p>

<button id="send">Отправить боту</button>

<script>
  // 1. Подключаем Telegram API
  const tg = window.Telegram.WebApp;
  tg.expand();

  // 2. Получаем данные пользователя
  const user = tg.initDataUnsafe.user;

  // 3. Проверяем, что данные есть
  if (!user) {
    alert("Открой Mini App через Telegram!");
  } else {
    // 4. Показываем данные на экране
    document.getElementById("uid").textContent = user.id;
    document.getElementById("uname").textContent = user.username || "нет";
  }

  // 5. Отправка данных боту
  document.getElementById("send").onclick = () => {
    tg.sendData(JSON.stringify({
      action: "user_info",
      user_id: user.id,
      username: user.username
    }));
  };
</script>
